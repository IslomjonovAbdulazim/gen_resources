Kontaktlar ro'yxati - ScrollController bilan
Vazifa: ListView.builder ishlatib, 100 ta kontakt ro'yxatini yarating. ScrollController qo'shing va "Yuqoriga" tugmasi orqali ro'yxat boshiga qaytish imkonini bering.
class ContactList extends StatefulWidget { @override _ContactListState createState() => _ContactListState(); } class _ContactListState extends State { ScrollController _scrollController = ScrollController(); @override Widget build(BuildContext context) { return Scaffold( body: ListView.builder( controller: _scrollController, itemCount: 100, itemBuilder: (context, index) { return Card( margin: EdgeInsets.symmetric(horizontal: 16, vertical: 4), child: ListTile( leading: CircleAvatar( backgroundColor: Colors.purple, child: Text('${String.fromCharCode(65 + (index % 26))}'), ), title: Text('Kontakt ${index + 1}'), subtitle: Text('+998 90 ${100 + index} ${200 + index}'), trailing: Icon(Icons.phone), onTap: () { print('Kontakt ${index + 1} ga qo\'ng\'iroq'); }, ), ); }, ), floatingActionButton: FloatingActionButton( onPressed: () { _scrollController.animateTo( 0, duration: Duration(milliseconds: 800), curve: Curves.easeInOut, ); }, child: Icon(Icons.keyboard_arrow_up), ), ); } @override void dispose() { _scrollController.dispose(); super.dispose(); } }
